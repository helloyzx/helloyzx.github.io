def clean(data,code):
    klines_list = [i.split(',') for i in data.get('klines')]
    df = pd.DataFrame(klines_list, columns=['date','open','close','high','low','vol','amount','turnover'])
    df['close'] = df['close'].astype(float)
    df['open'] = df['open'].astype(float)
    df['code'] = data['code']
    df['name'] = data['name']
    
    return df

def cleandf(df):
    # 당일수익률/익일수익률
    df.loc[:,'익일수익'] = df['close'].shift(-1).pct_change(1)
    df.loc[:,'당일수익'] = df['close'].pct_change(1)
    df.dropna(inplace=True)
    # 1은 상승, 0은 하락
    df.loc[df['익일수익'] > 0, '익일방향'] = 1
    df.loc[df['익일수익'] <= 0, '익일방향'] = 0
    df = df[['당일수익', '익일방향']]
    return df

def train_test(df):
    # 특징 X와 y 만들기
    y = df['익일방향'].values
    X = df.drop('익일방향', axis=1).values
    # train/test 구분
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.8, random_state=30)
    # 0.4test
    return X_train, X_test, y_train, y_test

def logi(X_train, X_test, y_train, y_test):
    
    logreg = LogisticRegression(solver='liblinear')

    # train
    logreg.fit(X_train, y_train)

    # test
    new_prediction = logreg.predict(X_test)

    #예측 맞는 개수/총 개수 
    return (logreg.score(X_test, y_test))

def svc(X_train, X_test, y_train, y_test):
    clf = svm.SVC(kernel = 'rbf', gamma='auto',C = 1000.0)
    clf.fit(X_train, y_train)
    new_prediction = clf.predict(X_test)
    return (clf.score(X_test, y_test))
